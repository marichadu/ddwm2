public function index() {
  $eventModel = new Event($this->db);

  // Navigation mensuelle : on lit ?year= et ?month= (défaut = mois courant)
  $year  = (int)($_GET['year']  ?? date('Y'));
  $month = (int)($_GET['month'] ?? date('n'));

  // Correction si la navigation dépasse les bornes 1-12
  if ($month < 1)  { $month = 12; $year--; }
  if ($month > 12) { $month = 1;  $year++; }

  // Filtres optionnels transmis via GET (formulaire <select>)
  $city     = trim($_GET['city']      ?? '');
  $gameType = trim($_GET['game_type'] ?? '');

  // Événements du mois, indexés par jour pour le rendu de la grille
  $monthEvents = $eventModel->getEventsByMonth(
      $year, $month, $city, $gameType
  );
  $eventsByDay = [];
  foreach ($monthEvents as $event) {
      $day = (int) date('j',
          strtotime($event['start_date']));
      $eventsByDay[$day][] = $event;
  }

  // Données pour les <select> de filtres
  $cities    = $eventModel->getCities();
  $gameTypes = $eventModel->getGameTypes();

  require_once '../app/views/events/index.php';
}

// Model Event.php - getEventsByMonth($year, $month, $city, $game_type)

$start = sprintf('%04d-%02d-01 00:00:00', $year, $month);
$end   = date('Y-m-t 23:59:59', mktime(0,0,0, $month,1,$year));

$conditions = ['e.start_date BETWEEN :start AND :end'];
$params     = [':start' => $start, ':end' => $end];

if ($city !== '') {
    $conditions[] = 'l.city = :city';
    $params[':city'] = $city;
}
if ($game_type !== '') {
    $conditions[] = 'e.game_type = :game_type';
    $params[':game_type'] = $game_type;
}

$where = 'WHERE ' . implode(' AND ', $conditions);

$stmt = $this->db->prepare("
    SELECT e.*, l.name AS location_name,
           l.city, l.address, l.maps_url, l.is_store,
           (SELECT COUNT(*) FROM registrations r
            WHERE r.event_id = e.id
              AND r.status = 'validated') AS confirmed_count
    FROM events e
    JOIN locations l ON e.location_id = l.id
    $where
    ORDER BY e.start_date ASC
");
$stmt->execute($params);
return $stmt->fetchAll(PDO::FETCH_ASSOC);

// Marque un joueur présent et crédite ses Templix (POST uniquement).
// Le manager coche les présents depuis la page événement le jour J.
public function checkin() {
  $this->requireManager();
  csrfValidate(BASE_URL . '?action=manager');

  $registration_id = (int)($_POST['registration_id'] ?? 0);
  $event_id        = (int)($_POST['event_id'] ?? 0);

  if ($registration_id <= 0 || $event_id <= 0) {
    header('Location: ' . BASE_URL . '?action=manager');
    exit;
  }

  $eventModel        = new Event($this->db);
  $registrationModel = new Registration($this->db);

  $event        = $eventModel->getEventById($event_id);
  $registration = $registrationModel->getUserRegistrationById($registration_id);

  // Validation : paiement requis si l'événement est payant
  if ((float)$event['price'] > 0 && $registration['paid'] != 1) {
    header("Location: " . BASE_URL . "?action=manager-event&id={$event_id}&msg=error-unpaid");
    exit;
  }

  $templix_reward = (int)($event['templix_reward'] ?? 0);

  // On a besoin de l'user_id pour créditer les Templix - on le lit depuis registrations.
  if ($registration) {
    $registrationModel->checkin($registration_id, $registration['user_id'], $templix_reward);
  }

  // Redirige vers le scanner si émargement via QR, sinon vers la page de l'événement
  $redirect = $_POST['redirect'] ?? '';
  if ($redirect === 'scan') {
    header('Location: ' . BASE_URL . '?action=manager-scan&msg=checked-in');
  } else {
    header("Location: " . BASE_URL . "?action=manager-event&id={$event_id}&msg=checked-in");
  }
  exit;
}

Model - Registration.php (transaction atomique MySQL via PDO)
public function checkin($registration_id, $user_id, $templix_reward = 0) {
  $this->db->beginTransaction();
  try {
    // On marque présent uniquement si pas déjà fait (checked_in IS NULL → pas encore traité)
    $stmt = $this->db->prepare("UPDATE registrations SET checked_in = 1
        WHERE id = :id AND checked_in IS NULL");
    $stmt->execute([':id' => $registration_id]);

    // rowCount() = 0 : déjà émargé - on ne crédite pas deux fois.
    if ($stmt->rowCount() === 0) {
      $this->db->rollBack();
      return false;
    }

    // Crédit Templix dans la même transaction - atomique avec le marquage.
    if ($templix_reward > 0) {
      $stmt = $this->db->prepare("UPDATE users
          SET templix_balance = templix_balance + :amount WHERE id = :id");
      $stmt->execute([':amount' => $templix_reward, ':id' => $user_id]);
    }

    $this->db->commit();
    return true;

  } catch (Exception $e) {
    $this->db->rollBack();
    throw $e;
  }
}

Sécurité

Ce qu'on a vu dans le code

  • CSRF - csrfField() dans le formulaire + hash_equals() côté Controller
  • Contrôle d'accès - requireManager() vérifie la session avant chaque action
  • Transaction atomique - émargement + crédit Templix dans un beginTransaction / commit
  • Idempotence - WHERE checked_in IS NULL empêche le double-checkin
  • XSS - htmlspecialchars() sur toutes les sorties dans la vue
  • Casting - (int) sur les IDs POST pour éviter l'injection

Ailleurs dans le projet

  • Mots de passe - password_hash(BCRYPT) + password_verify()
  • Sessions - session_regenerate_id(true) après login pour éviter le fixation
  • SQL - 100 % requêtes préparées PDO, zéro concaténation
  • Reset password - token random_bytes(32), expiration 1 h, usage unique
